1. Query executada (RCP-009 — aproveitamento últimos N)

WITH res AS (
  SELECT team_id_home AS tid, team_name_home AS nome, game_date, wl_home AS wl
  FROM game WHERE season_type='Regular Season'
  UNION ALL
  SELECT team_id_away, team_name_away, game_date, wl_away
  FROM game WHERE season_type='Regular Season'
),
r AS (
  SELECT tid, nome, wl,
         ROW_NUMBER() OVER (PARTITION BY tid ORDER BY game_date DESC) AS rn
  FROM res WHERE tid IN (SELECT id FROM team)
)
SELECT t.full_name AS time, COUNT(*) AS jogos,
       COUNT(*) FILTER (WHERE r.wl='W') AS vitorias,
       ROUND(COUNT(*) FILTER (WHERE r.wl='W')*1.0/COUNT(*),3) AS aproveitamento
FROM r JOIN team t ON t.id=r.tid
WHERE r.rn<=10
GROUP BY t.full_name
ORDER BY aproveitamento DESC, vitorias DESC;
